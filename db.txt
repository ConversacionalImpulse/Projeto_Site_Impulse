model Employee {
  id                Int       @id @default(autoincrement())
  //Dados Pessoais
  nome              String
  rg                String?   @unique
  cpf               String    @unique
  dataNascimento    DateTime
  email             String    @unique
  telefoneWapp      String
  salario           String?
  codBanco          String?
  agenciaBanco      String?
  contaBanco        String?
  pixBanco          String?
  dataContratacao   DateTime
  modeloContratacao String
  departamento      String
  cargo             String
  estagio           String?
  escolaVinculo     String?
  horasAlocadas     Int
  dataDemissao      DateTime?
  turnOver          String?
  formulario1       Form1[]
  formulario2       Form2[]
  address           Address?

  @@map("funcionarios")
}

model Address {
  id          Int     @id @default(autoincrement())
  rua         String
  numero      String
  complemento String?
  bairro      String
  cidade      String
  estado      String
  cep         String

  fk_employeeId Int      @unique
  employee      Employee @relation(fields: [fk_employeeId], references: [id])

  @@map("endereco")
}

  bairro              String?
  cidade              String?
  emailResponsavel    String
  nomeResponsavel     String
  telefoneResponsavel String?
